sig
  type phrase = string list
  type command =
      Quit
    | Help
    | Castle of Command.phrase
    | Move of Command.phrase
    | Start of Command.phrase
    | View of Command.phrase
    | Back
    | Next
    | Make of Command.phrase
    | Checkmate
  exception InvalidCommand of string
  exception InvalidRow of string
  exception InvalidColumn of string
  val parse : string -> Command.command
end
